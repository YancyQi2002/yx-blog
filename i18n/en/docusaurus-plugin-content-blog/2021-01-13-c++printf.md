## 01 实现打印字符串

```c++
#include <stdio.h>

int my_printf(const char *frm, ...) {
	int cnt = 0;
	#define PUTC(a) putchar(a), ++cnt
	for (int i = 0; frm[i]; i ++) {
	//	putchar(frm[i]),++cnt;
		PUTC(frm[i]);
	}
	return cnt;
}

int main() {
	printf("hello world\n");
	my_printf("hello world\n");
	return 0;
}
```
